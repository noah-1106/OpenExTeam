// 启动任务 - 发送任务消息给对应Agent function startTask(taskId) { var task = _tasks.find(function(t) { return t.id === taskId; }); if (!task) return; // 模拟任务下发 var msg = "【任务启动】" + task.title; if (task.description) msg += "\n描述：" + task.description; if (task.excard) msg += "\nExCard：" + task.excard; // 找到对应Agent的会话 var agentName = task.agent; var sess = SESSIONS.find(function(s) { return s.type === "p2p" && s.name === agentName; }); if (sess) { sess.messages.push({ sender: "user", text: msg, time: new Date().toLocaleTimeString("zh-CN", {hour: "2-digit", minute: "2-digit"}) }); // 模拟Agent回复 setTimeout(function() { sess.messages.push({ sender: agentName, agentId: sess.agentId, text: "收到任务：" + task.title + "，开始执行...", time: new Date().toLocaleTimeString("zh-CN", {hour: "2-digit", minute: "2-digit"}) }); // 更新任务状态为进行中 task.status = "in-progress"; renderTab(); }, 800); alert("任务已启动，消息已发送给 " + agentName); closeDetail(); renderTab(); } else { alert("未找到 " + agentName + " 的聊天会话"); } } // ===== WORKFLOW ===== function startJobWorkflow(jobId) { var job = _jobs.find(function(j) { return j.id === jobId; }); if (!job) return; // 找到该工作的所有任务，按创建时间排序 var jobTasks = _tasks.filter(function(t) { return t.jobId === jobId; }); jobTasks.sort(function(a, b) { return new Date(a.createdAt) - new Date(b.createdAt); }); if (jobTasks.length === 0) { alert("该工作没有关联任务"); return; } // 找到第一个待办的任务 var firstTodo = jobTasks.find(function(t) { return t.status === "todo"; }); if (!firstTodo) { // 如果没有待办任务，找第一个进行中的 firstTodo = jobTasks.find(function(t) { return t.status === "in-progress"; }); if (!firstTodo) { alert("该工作所有任务已完成"); return; } } // 发送消息给对应Agent var agentName = firstTodo.agent; var sess = SESSIONS.find(function(s) { return s.type === "p2p" && s.name === agentName; }); if (sess) { var stepNum = jobTasks.indexOf(firstTodo) + 1; var msg = "【工作流启动】" + job.title + "\n"; msg += "步骤 #" + stepNum + "/" + jobTasks.length + "：" + firstTodo.title; if (firstTodo.description) msg += "\n" + firstTodo.description; sess.messages.push({ sender: "user", text: msg, time: new Date().toLocaleTimeString("zh-CN", {hour: "2-digit", minute: "2-digit"}) }); // 模拟Agent回复 setTimeout(function() { sess.messages.push({ sender: agentName, agentId: sess.agentId, text: "收到工作流任务，开始执行步骤 #" + stepNum + "...", time: new Date().toLocaleTimeString("zh-CN", {hour: "2-digit", minute: "2-digit"}) }); // 更新任务状态为进行中 firstTodo.status = "in-progress"; job.status = "in-progress"; renderTab(); }, 800); alert("工作流已启动，步骤 #" + stepNum + " 已发送给 " + agentName); renderTab(); } else { alert("未找到 " + agentName + " 的聊天会话"); } } // 工作流步骤完成回调（Adapter会调用此函数） function onWorkflowStepComplete(jobId, taskId) { var job = _jobs.find(function(j) { return j.id === jobId; }); var completedTask = _tasks.find(function(t) { return t.id === taskId; }); if (!job || !completedTask) return; // 标记当前任务为完成 completedTask.status = "done"; // 找到该工作的所有任务，按创建时间排序 var jobTasks = _tasks.filter(function(t) { return t.jobId === jobId; }); jobTasks.sort(function(a, b) { return new Date(a.createdAt) - new Date(b.createdAt); }); // 找到下一个待办任务 var nextTask = jobTasks.find(function(t) { return t.status === "todo"; }); if (nextTask) { // 自动启动下一步 var stepNum = jobTasks.indexOf(nextTask) + 1; var agentName = nextTask.agent; var sess = SESSIONS.find(function(s) { return s.type === "p2p" && s.name === agentName; }); if (sess) { var msg = "【工作流继续】" + job.title + "\n"; msg += "步骤 #" + stepNum + "/" + jobTasks.length + "：" + nextTask.title; if (nextTask.description) msg += "\n" + nextTask.description; sess.messages.push({ sender: "user", text: msg, time: new Date().toLocaleTimeString("zh-CN", {hour: "2-digit", minute: "2-digit"}) }); setTimeout(function() { sess.messages.push({ sender: agentName, agentId: sess.agentId, text: "收到工作流任务，开始执行步骤 #" + stepNum + "...", time: new Date().toLocaleTimeString("zh-CN", {hour: "2-digit", minute: "2-digit"}) }); nextTask.status = "in-progress"; renderTab(); }, 800); } } else { // 所有任务完成 job.status = "done"; // 通知所有参与者 jobTasks.forEach(function(t) { var sess = SESSIONS.find(function(s) { return s.type === "p2p" && s.name === t.agent; }); if (sess) { sess.messages.push({ sender: "user", text: "【工作流完成】" + job.title + " 所有步骤已完成！", time: new Date().toLocaleTimeString("zh-CN", {hour: "2-digit", minute: "2-digit"}) }); } }); renderTab(); } }
